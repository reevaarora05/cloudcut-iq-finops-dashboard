1. Cloud Waste Formula

def cloud_waste(cloud_spend, industry_mult, size_mult, provider_factor):
    """Estimate annual cloud waste in dollars."""
    base_pct = 0.295
    adjustment = (industry_mult - 1) * 0.08 + (provider_factor - 1) * 0.06
    adjustment -= (size_mult - 1) * 0.02
    waste_pct = min(max(base_pct + adjustment, 0.27), 0.32)
    return cloud_spend * waste_pct, waste_pct

2. SaaS Utilization Scoring

def saas_unused_pct(saas_tools, size_mult):
    """Estimate the share of paid SaaS licenses that go unused."""
    tool_factor = (saas_tools - 20) / 480 * 0.15
    size_factor = (size_mult - 1) * 0.05
    unused_pct = 0.35 + tool_factor + size_factor
    return min(max(unused_pct, 0.35), 0.55)

def saas_waste_breakdown(annual_saas_spend, unused_pct):
    total_waste = annual_saas_spend * unused_pct
    return {
        "unused_licenses": total_waste * 0.60,
        "duplicate_tools": total_waste * 0.25,
        "auto_renewed":    total_waste * 0.10,
        "shadow_it":       total_waste * 0.05,
    }

3. Renewal Risk Scoring

def renewal_risk(days_until_renewal, auto_renew, annual_value, avg_value):
    """Classify a contract's renewal risk as High, Medium, or Low."""
    score = 0
    if days_until_renewal <= 30:
        score += 2
    elif days_until_renewal <= 90:
        score += 1

    if auto_renew:
        score += 1

    if annual_value > avg_value * 1.5:
        score += 1

    if score >= 3:
        return "High"
    elif score >= 1:
        return "Medium"
    return "Low"
